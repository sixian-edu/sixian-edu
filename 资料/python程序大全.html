from string import ascii_uppercase as uppers
from string import ascii_letters as letters

39 内部函数实现组合数 C(m,n)

def cmn(m, n):             # 外部函数
    def fac_demo(n):      # 内部函数：求阶乘
        f = 1
        for i in range(1, n + 1):
            f *= i
        return f
    return fac_demo(m) / (fac_demo(n) * fac_demo(m - n))

print('c(5,2)=', cmn(5, 2))  # 10.0

def AllPrime(m):     # 求所有素因子
    def IsPrime(n):  # 判断一个数是否是素数
        return 0 not in [0 for i in range(2, n) if n % i == 0]
    return [i for i in range(2, m) if m % i == 0 and IsPrime(i)]

m = int(input('enter a number:'))
print('{}的素因子有{}'.format(m, AllPrime(m)))

class Vector:
    def __init__(self, a, b):
        self.__a = a
        self.__b = b
    def out(self):
        print('Vector({},{})'.format(self.__a, self.__b))
    def __add__(self, other):          # 重载 + 运算符
        return Vector(self.__a + other.__a, self.__b + other.__b)
    def __repr__(self):                # 重载字符串表示
        return 'This is a Vector'
    __str__ = __repr__

v1 = Vector(1, 2)
